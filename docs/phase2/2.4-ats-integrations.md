# Phase 2.4: ATS Direct Integrations

**Status:** Proposed

**Owner:** Backend

## Problem Statement

Currently we rely on Apify's LinkedIn scraper for job data:
- **Coverage:** Only ~50% of companies (those advertising on LinkedIn)
- **Freshness:** Depends on Apify crawl schedule (daily/weekly)
- **Data Quality:** Scraped data is less structured than API data
- **Reliability:** LinkedIn can block scrapers, change HTML structure

**Opportunity:** 50.5% of companies use identifiable ATS systems that we can query directly:
- Greenhouse: 9 companies
- Lever: 12 companies
- Ashby: 5 companies
- JobAdder: 15 companies
- Employment Hero: 51 companies
- SnapHire: 2 companies

---

```kotlin
package com.techmarket.sync.ats

import java.time.Instant

/**
 * Common interface for all ATS providers.
 */
interface AtsProvider {
    val name: String
    val priority: Int  // For merge conflicts (higher = more authoritative)
    
    /**
     * Fetches all jobs for a company using this ATS.
     */
    suspend fun fetchJobs(companyToken: String): List<AtsJob>
    
    /**
     * Fetches job details for a specific job ID.
     */
    suspend fun fetchJobDetails(companyToken: String, jobId: String): AtsJob?
    
    /**
     * Tests if the ATS is accessible and credentials are valid.
     */
    suspend fun healthCheck(companyToken: String): AtsHealthStatus
}

/**
 * Unified job representation from any ATS.
 */
data class AtsJob(
    val id: String,
    val title: String,
    val companyId: String,
    val companyName: String,
    val description: String,
    val locations: List<AtsLocation>,
    val department: String?,
    val employmentType: String?,      // FULL_TIME, PART_TIME, CONTRACT, INTERN
    val seniorityLevel: String?,      // JUNIOR, MID, SENIOR, LEAD
    val salaryMin: Long?,             // In smallest currency unit (cents)
    val salaryMax: Long?,
    val salaryCurrency: String?,      // NZD, AUD, USD
    val salaryPeriod: String?,        // HOUR, DAY, MONTH, YEAR
    val benefits: List<String>,
    val requirements: List<String>,
    val applicationUrl: String,
    val postedAt: Instant,
    val closedAt: Instant?,
    val remote: Boolean,
    val source: String                // ATS provider name
)

data class AtsLocation(
    val city: String?,
    val state: String?,
    val country: String,
    val remote: Boolean
)

data class AtsHealthStatus(
    val healthy: Boolean,
    val message: String,
    val lastChecked: Instant
)
```

```kotlin
package com.techmarket.sync.ats.greenhouse

import com.techmarket.sync.ats.*
import org.springframework.http.MediaType
import org.springframework.stereotype.Component
import org.springframework.web.reactive.function.client.WebClient
import org.springframework.web.reactive.function.client.bodyToMono
import java.time.Instant

@Component
class GreenhouseApiClient(
    private val webClient: WebClient
) : AtsProvider {
    
    override val name = "GREENHOUSE"
    override val priority = 80  // High priority - official source
    
    private val baseUrl = "https://boards-api.greenhouse.io/v1/boards"
    
    override suspend fun fetchJobs(companyToken: String): List<AtsJob> {
        val url = "$baseUrl/$companyToken/jobs"
        
        val response = webClient.get()
            .uri(url)
            .accept(MediaType.APPLICATION_JSON)
            .retrieve()
            .bodyToMono<GreenhouseJobsResponse>()
            .await()
        
        return response.jobs.map { job ->
            job.toAtsJob(companyToken)
        }
    }
    
    override suspend fun fetchJobDetails(
        companyToken: String,
        jobId: String
    ): AtsJob? {
        val url = "$baseUrl/$companyToken/jobs/$jobId"
        
        return try {
            webClient.get()
                .uri(url)
                .accept(MediaType.APPLICATION_JSON)
                .retrieve()
                .bodyToMono<GreenhouseJobDetail>()
                .await()
                ?.toAtsJob(companyToken)
        } catch (e: Exception) {
            null  // Job not found or error
        }
    }
    
    override suspend fun healthCheck(companyToken: String): AtsHealthStatus {
        return try {
            val jobs = fetchJobs(companyToken)
            AtsHealthStatus(
                healthy = true,
                message = "Successfully fetched ${jobs.size} jobs",
                lastChecked = Instant.now()
            )
        } catch (e: Exception) {
            AtsHealthStatus(
                healthy = false,
                message = "Health check failed: ${e.message}",
                lastChecked = Instant.now()
            )
        }
    }
}

// Data classes for Greenhouse API responses

data class GreenhouseJobsResponse(
    val jobs: List<GreenhouseJob>
)

data class GreenhouseJob(
    val id: Long,
    val title: String,
    val location: GreenhouseLocation?,
    val departments: List<GreenhouseDepartment>?,
    val offices: List<GreenhouseOffice>?,
    val updated_at: String
)

data class GreenhouseJobDetail(
    val id: Long,
    val title: String,
    val description: String,
    val location: GreenhouseLocation?,
    val departments: List<GreenhouseDepartment>?,
    val offices: List<GreenhouseOffice>?,
    val updated_at: String,
    val requirements: List<GreenhouseRequirement>?,
    val application_url: String
)

data class GreenhouseLocation(
    val name: String
)

data class GreenhouseDepartment(
    val name: String
)

data class GreenhouseOffice(
    val name: String
)

data class GreenhouseRequirement(
    val label: String,
    val description: String?
)

// Extension functions to convert Greenhouse types to AtsJob

fun GreenhouseJob.toAtsJob(companyToken: String): AtsJob {
    return AtsJob(
        id = id.toString(),
        title = title,
        companyId = companyToken,  // Will be mapped to actual company ID later
        companyName = companyToken,  // Will be resolved
        description = "",  // Need to fetch details for full description
        locations = listOfNotNull(location?.toAtsLocation()),
        department = departments?.firstOrNull()?.name,
        employmentType = null,  // Not provided in list endpoint
        seniorityLevel = null,
        salaryMin = null,
        salaryMax = null,
        salaryCurrency = null,
        salaryPeriod = null,
        benefits = emptyList(),
        requirements = emptyList(),
        applicationUrl = "https://boards.greenhouse.io/$companyToken/jobs/$id",
        postedAt = Instant.parse(updatedAt),
        closedAt = null,
        remote = location?.name?.contains("Remote") == true,
        source = "GREENHOUSE"
    )
}

fun GreenhouseLocation.toAtsLocation(): AtsLocation {
    // Parse location string like "Sydney, Australia" or "Remote - Australia"
    val parts = name.split(",").map { it.trim() }
    return AtsLocation(
        city = parts.getOrNull(0),
        state = null,  // Greenhouse doesn't provide state separately
        country = parts.getOrNull(1) ?: "Unknown",
        remote = name.contains("Remote", ignoreCase = true)
    )
}
```

```kotlin
package com.techmarket.sync.ats.lever

import com.techmarket.sync.ats.*
import org.springframework.stereotype.Component
import org.springframework.web.reactive.function.client.WebClient
import java.time.Instant

@Component
class LeverApiClient(
    private val webClient: WebClient
) : AtsProvider {
    
    override val name = "LEVER"
    override val priority = 80
    
    private val baseUrl = "https://api.lever.co/v0/postings"
    
    override suspend fun fetchJobs(companyToken: String): List<AtsJob> {
        val url = "$baseUrl/$companyToken"
        
        val response = webClient.get()
            .uri(url)
            .retrieve()
            .bodyToMono<List<LeverPosting>>()
            .await()
        
        return response.map { posting ->
            posting.toAtsJob(companyToken)
        }
    }
    
    override suspend fun fetchJobDetails(
        companyToken: String,
        jobId: String
    ): AtsJob? {
        // Lever doesn't have a single job endpoint, use list and filter
        return fetchJobs(companyToken).find { it.id == jobId }
    }
    
    override suspend fun healthCheck(companyToken: String): AtsHealthStatus {
        return try {
            val jobs = fetchJobs(companyToken)
            AtsHealthStatus(
                healthy = true,
                message = "Successfully fetched ${jobs.size} jobs",
                lastChecked = Instant.now()
            )
        } catch (e: Exception) {
            AtsHealthStatus(
                healthy = false,
                message = "Health check failed: ${e.message}",
                lastChecked = Instant.now()
            )
        }
    }
}

data class LeverPosting(
    val id: String,
    val text: String,
    val title: String,
    val department: String,
    val location: String,
    val list: String,  // "open" or "closed"
    val createdAt: Long,  // Unix timestamp in milliseconds
    val description: String,
    val categories: LeverCategories
)

data class LeverCategories(
    val commitment: List<LeverCommitment>,
    val level: List<LeverLevel>
)

data class LeverCommitment(
    val text: String  // "Full-Time", "Part-Time", "Contract"
)

data class LeverLevel(
    val text: String  // "Junior", "Mid-Level", "Senior"
)

fun LeverPosting.toAtsJob(companyToken: String): AtsJob {
    return AtsJob(
        id = id,
        title = title,
        companyId = companyToken,
        companyName = companyToken,
        description = description,
        locations = listOf(
            AtsLocation(
                city = null,
                state = null,
                country = location,
                remote = location.contains("Remote", ignoreCase = true)
            )
        ),
        department = department,
        employmentType = categories.commitment.firstOrNull()?.text?.uppercase(),
        seniorityLevel = categories.level.firstOrNull()?.text?.uppercase(),
        salaryMin = null,  // Lever doesn't provide salary in public API
        salaryMax = null,
        salaryCurrency = null,
        salaryPeriod = null,
        benefits = emptyList(),
        requirements = emptyList(),
        applicationUrl = "https://apply.lever.co/postings/$id",
        postedAt = Instant.ofEpochMilli(createdAt),
        closedAt = if (list == "closed") Instant.now() else null,
        remote = location.contains("Remote", ignoreCase = true),
        source = "LEVER"
    )
}
```

### Step 4: Create ATS Sync Service (3 hours)

**File:** `backend/src/main/kotlin/com/techmarket/sync/ats/AtsSyncService.kt`

```kotlin
package com.techmarket.sync.ats

import com.techmarket.persistence.company.CompanyRepository
import com.techmarket.persistence.job.JobRepository
import com.techmarket.sync.RawJobDataMapper
import kotlinx.coroutines.*
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import java.time.Instant

@Service
class AtsSyncService(
    private val greenhouseClient: GreenhouseApiClient,
    private val leverClient: LeverApiClient,
    // Add more clients as implemented
    private val companyRepository: CompanyRepository,
    private val jobRepository: JobRepository,
    private val rawJobDataMapper: RawJobDataMapper
) {
    
    private val log = LoggerFactory.getLogger(AtsSyncService::class.java)
    
    private val providers: Map<String, AtsProvider> by lazy {
        mapOf(
            "GREENHOUSE" to greenhouseClient,
            "LEVER" to leverClient
        )
    }
    
    /**
     * Syncs jobs for all companies with known ATS providers.
     */
    suspend fun syncAllCompanies(): AtsSyncSummary {
        val companies = companyRepository.getAllCompaniesWithAts()
        log.info("Starting ATS sync for ${companies.size} companies")
        
        val results = mutableListOf<AtsCompanyResult>()
        
        coroutineScope {
            companies.map { company ->
                launch {
                    try {
                        val result = syncCompany(company)
                        results.add(result)
                    } catch (e: Exception) {
                        log.error("Failed to sync company ${company.id}: ${e.message}")
                        results.add(
                            AtsCompanyResult(
                                companyId = company.id,
                                provider = company.atsProvider,
                                success = false,
                                jobsFetched = 0,
                                error = e.message
                            )
                        )
                    }
                }
            }
        }
        
        return AtsSyncSummary(
            totalCompanies = companies.size,
            successfulSyncs = results.count { it.success },
            failedSyncs = results.count { !it.success },
            totalJobsFetched = results.sumOf { it.jobsFetched },
            results = results
        )
    }
    
    /**
     * Syncs jobs for a single company.
     */
    private suspend fun syncCompany(company: CompanyRecord): AtsCompanyResult {
        val provider = providers[company.atsProvider]
            ?: return AtsCompanyResult(
                companyId = company.id,
                provider = company.atsProvider,
                success = false,
                jobsFetched = 0,
                error = "Unknown ATS provider: ${company.atsProvider}"
            )
        
        log.info("Syncing ${company.name} via ${provider.name}")
        
        val atsJobs = provider.fetchJobs(company.atsToken)
        log.info("Fetched ${atsJobs.size} jobs for ${company.name}")
        
        // Convert ATS jobs to our internal format
        val jobRecords = atsJobs.map { atsJob ->
            rawJobDataMapper.mapAtsJobToRecord(atsJob, company)
        }
        
        // Upsert jobs to BigQuery
        jobRepository.upsertJobs(jobRecords)
        
        return AtsCompanyResult(
            companyId = company.id,
            provider = company.atsProvider,
            success = true,
            jobsFetched = atsJobs.size,
            error = null
        )
    }
    
    /**
     * Runs health checks on all ATS providers.
     */
    suspend fun healthCheckAll(): Map<String, AtsHealthStatus> {
        val companies = companyRepository.getAllCompaniesWithAts()
        val results = mutableMapOf<String, AtsHealthStatus>()
        
        for (company in companies) {
            val provider = providers[company.atsProvider] ?: continue
            val status = provider.healthCheck(company.atsToken)
            results["${company.name} (${company.atsProvider})"] = status
        }
        
        return results
    }
    
    data class AtsSyncSummary(
        val totalCompanies: Int,
        val successfulSyncs: Int,
        val failedSyncs: Int,
        val totalJobsFetched: Int,
        val results: List<AtsCompanyResult>
    )
    
    data class AtsCompanyResult(
        val companyId: String,
        val provider: String,
        val success: Boolean,
        val jobsFetched: Int,
        val error: String?
    )
}
```

```kotlin
package com.techmarket.scheduler

import com.techmarket.service.AtsSyncService
import kotlinx.coroutines.runBlocking
import org.slf4j.LoggerFactory
import org.springframework.scheduling.annotation.Scheduled
import org.springframework.stereotype.Component

@Component
class AtsSyncScheduler(
    private val atsSyncService: AtsSyncService
) {
    
    private val log = LoggerFactory.getLogger(AtsSyncScheduler::class.java)
    
    /**
     * Runs daily at 1:00 AM UTC (1:00 PM NZST / 11:00 AM AEST).
     * Before the main Apify sync to ensure fresh data.
     */
    @Scheduled(cron = "0 0 1 * * *", zone = "UTC")
    fun runDailyAtsSync() {
        log.info("Starting scheduled daily ATS sync")
        
        try {
            runBlocking {
                val summary = atsSyncService.syncAllCompanies()
                log.info("ATS sync completed: ${summary.successfulSyncs}/${summary.totalCompanies} companies, " +
                        "${summary.totalJobsFetched} jobs fetched")
            }
        } catch (e: Exception) {
            log.error("ATS sync failed: ${e.message}", e)
        }
    }
}
```

```kotlin
@RestController
@RequestMapping("/api/admin")
class AdminController(
    private val atsSyncService: AtsSyncService
) {
    
    @PostMapping("/ats/sync")
    suspend fun triggerAtsSync(): ResponseEntity<Map<String, Any>> {
        val summary = atsSyncService.syncAllCompanies()
        
        return ResponseEntity.ok(mapOf(
            "status" to "completed",
            "summary" to mapOf(
                "totalCompanies" to summary.totalCompanies,
                "successfulSyncs" to summary.successfulSyncs,
                "failedSyncs" to summary.failedSyncs,
                "totalJobsFetched" to summary.totalJobsFetched
            )
        ))
    }
    
    @GetMapping("/ats/health")
    suspend fun getAtsHealth(): ResponseEntity<Map<String, Any>> {
        val healthResults = atsSyncService.healthCheckAll()
        
        return ResponseEntity.ok(mapOf(
            "providers" to healthResults.map { (name, status) ->
                mapOf(
                    "name" to name,
                    "healthy" to status.healthy,
                    "message" to status.message,
                    "lastChecked" to status.lastChecked.toString()
                )
            }
        ))
    }
}
```
