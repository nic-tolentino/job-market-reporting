No visibility into what's hot - most viewed jobs/companies/tech in the last 7 days.

- User engagement (FOMO effect)
- Discovery of popular content
- Market insights

```typescript
import { track } from '@vercel/analytics';

export function trackPageView(pageType: string, pageId: string) {
    track('page_view', {
        pageType,  // 'JOB', 'COMPANY', 'TECH'
        pageId,
        timestamp: new Date().toISOString()
    });
}

trackPageView('TECH', 'react');
trackPageView('JOB', '12345');
```

### Step 2: Create Vercel Analytics Service (2 hours)

**File:** `backend/src/main/kotlin/com/techmarket/service/VercelAnalyticsService.kt`

```kotlin
@Service
class VercelAnalyticsService(
    @Value("\${vercel.analytics.team-id}")
    private val teamId: String,

    @Value("\${vercel.analytics.api-token}")
    private val apiToken: String,

    private val webClient: WebClient
) {

    data class PageViewEvent(
        val timestamp: Long,
        val pageType: String,
        val pageId: String,
        val country: String?
    )

    suspend fun getPageViews(
        startTime: Instant,
        endTime: Instant,
        pageType: String? = null
    ): List<PageViewEvent> {
        val url = "https://analytics.vercel.com/api/v2/analytics"
        
        val response = webClient.get()
            .uri(url) {
                it.param("teamId", teamId)
                it.param("start", startTime.epochSecond.toString())
                it.param("end", endTime.epochSecond.toString())
                if (pageType != null) {
                    it.param("pageType", pageType)
                }
                it.build()
            }
            .header("Authorization", "Bearer $apiToken")
            .retrieve()
            .bodyToMono<VercelAnalyticsResponse>()
            .await()

        return response.events.map { it.toPageViewEvent() }
    }

    suspend fun getTrendingCompanies(limit: Int = 10): List<TrendingItem> {
        val now = Instant.now()
        val sevenDaysAgo = now.minus(7, ChronoUnit.DAYS)

        val pageViews = getPageViews(sevenDaysAgo, now, "COMPANY")
        
        return pageViews
            .groupingBy { it.pageId }
            .eachCount()
            .entries
            .sortedByDescending { it.value }
            .take(limit)
            .map { TrendingItem(it.key, it.value) }
    }

    suspend fun getTrendingTechnologies(limit: Int = 10): List<TrendingItem> {
        val now = Instant.now()
        val sevenDaysAgo = now.minus(7, ChronoUnit.DAYS)

        val pageViews = getPageViews(sevenDaysAgo, now, "TECH")
        
        return pageViews
            .groupingBy { it.pageId }
            .eachCount()
            .sortedByDescending { it.value }
            .take(limit)
            .map { TrendingItem(it.key, it.value) }
    }

    suspend fun getTrendingJobs(limit: Int = 10): List<TrendingItem> {
        val now = Instant.now()
        val sevenDaysAgo = now.minus(7, ChronoUnit.DAYS)

        val pageViews = getPageViews(sevenDaysAgo, now, "JOB")
        
        return pageViews
            .groupingBy { it.pageId }
            .eachCount()
            .sortedByDescending { it.value }
            .take(limit)
            .map { TrendingItem(it.key, it.value) }
    }
}

data class TrendingItem(
    val id: String,
    val viewCount: Int
)
```

### Step 3: Create Trending Service (1 hour)

**File:** `backend/src/main/kotlin/com/techmarket/service/TrendingService.kt`

```kotlin
@Service
class TrendingService(
    private val vercelAnalytics: VercelAnalyticsService,
    private val companyRepository: CompanyRepository,
    private val techRepository: TechRepository,
    private val jobRepository: JobRepository
) {

    suspend fun getTrendingCompanies(country: String?, limit: Int = 10): List<TrendingCompanyDto> {
        val trending = vercelAnalytics.getTrendingCompanies(limit * 2)  // Get extra for filtering
        
        val companies = companyRepository.getCompaniesByIds(trending.map { it.id })
        
        return trending
            .mapNotNull { trend ->
                companies.find { it.id == trend.id }?.let { company ->
                    TrendingCompanyDto(
                        company = company,
                        viewCount = trend.viewCount,
                        rank = trending.indexOf(trend) + 1
                    )
                }
            }
            .filter { country == null || it.company.operatingCountries.contains(country) }
            .take(limit)
    }

    suspend fun getTrendingTechnologies(country: String?, limit: Int = 10): List<TrendingTechDto> {
        val trending = vercelAnalytics.getTrendingTechnologies(limit * 2)
        
        return trending
            .map { trend ->
                TrendingTechDto(
                    name = TechFormatter.format(trend.id),
                    viewCount = trend.viewCount,
                    rank = trending.indexOf(trend) + 1
                )
            }
            .take(limit)
    }
}

data class TrendingCompanyDto(
    val company: CompanyDto,
    val viewCount: Int,
    val rank: Int
)

data class TrendingTechDto(
    val name: String,
    val viewCount: Int,
    val rank: Int
)
```

### Step 4: Add API Endpoint (1 hour)

**File:** `backend/src/main/kotlin/com/techmarket/api/TrendingController.kt`

```kotlin
@RestController
@RequestMapping("/api/trending")
class TrendingController(
    private val trendingService: TrendingService
) {

    @GetMapping("/companies")
    suspend fun getTrendingCompanies(
        @RequestParam country: String? = null,
        @RequestParam limit: Int = 10
    ): ResponseEntity<List<TrendingCompanyDto>> {
        val trending = trendingService.getTrendingCompanies(country, limit)
        return ResponseEntity.ok(trending)
    }

    @GetMapping("/technologies")
    suspend fun getTrendingTechnologies(
        @RequestParam country: String? = null,
        @RequestParam limit: Int = 10
    ): ResponseEntity<List<TrendingTechDto>> {
        val trending = trendingService.getTrendingTechnologies(country, limit)
        return ResponseEntity.ok(trending)
    }

    @GetMapping("/jobs")
    suspend fun getTrendingJobs(
        @RequestParam country: String? = null,
        @RequestParam limit: Int = 10
    ): ResponseEntity<List<TrendingJobDto>> {
        val trending = trendingService.getTrendingJobs(country, limit)
        return ResponseEntity.ok(trending)
    }
}
```
