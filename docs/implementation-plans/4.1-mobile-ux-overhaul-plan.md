export function Navbar() {
    const [scrolled, setScrolled] = useState(false);
    const [lastScrollY, setLastScrollY] = useState(0);
    const [visible, setVisible] = useState(true);

    // Smooth show/hide on scroll
    useEffect(() => {
        const handleScroll = () => {
            const currentScrollY = window.scrollY;
            
            if (currentScrollY > lastScrollY && currentScrollY > 100) {
                // Scrolling down - hide navbar
                setVisible(false);
            } else {
                // Scrolling up - show navbar
                setVisible(true);
            }
            
            setLastScrollY(currentScrollY);
            setScrolled(currentScrollY > 50);
        };

        window.addEventListener('scroll', handleScroll, { passive: true });
        return () => window.removeEventListener('scroll', handleScroll);
    }, [lastScrollY]);

    return (
        <nav className={`
            fixed 
            top-0 
            left-0 
            right-0 
            z-50
            bg-card/95 
            backdrop-blur-md
            border-b 
            border-border
            transition-transform 
            duration-300
            ${visible ? 'translate-y-0' : '-translate-y-full'}
            ${scrolled ? 'shadow-theme-sm' : ''}
        `}>
            <div className="max-w-7xl mx-auto px-4">
                <div className="flex items-center justify-between h-16 md:h-20">
                    {/* Logo */}
                    <Link to="/" className="flex-shrink-0">
                        <Logo className="h-8 md:h-10" />
                    </Link>

                    {/* Desktop Navigation */}
                    <div className="hidden md:flex items-center gap-6">
                        {/* Desktop menu items */}
                    </div>

                    {/* Mobile Menu Button - 44px min */}
                    <button
                        onClick={() => setMobileMenuOpen(!mobileMenuOpen)}
                        className="
                            md:hidden
                            p-2
                            rounded-lg
                            text-secondary
                            hover:text-primary
                            mobile:min-h-[44px]
                            mobile:min-w-[44px]
                        "
                        aria-label="Toggle menu"
                    >
                        {mobileMenuOpen ? (
                            <X className="w-6 h-6" />
                        ) : (
                            <Menu className="w-6 h-6" />
                        )}
                    </button>

                    {/* Right Side Actions */}
                    <div className="hidden md:flex items-center gap-4">
                        <CountrySelector />
                        <ThemeToggle />
                        <Link to="/about">About</Link>
                    </div>
                </div>
            </div>

            {/* Mobile Menu */}
            {mobileMenuOpen && (
                <div className="md:hidden border-t border-border bg-card">
                    <div className="px-4 py-4 space-y-4">
                        {/* Country Selector - Full width on mobile */}
                        <div className="pb-4 border-b border-border">
                            <CountrySelector />
                        </div>

                        {/* Menu Items - Larger tap targets */}
                        <Link 
                            to="/"
                            className="block py-3 text-base mobile:min-h-[44px]"
                        >
                            Home
                        </Link>
                        <Link 
                            to="/companies"
                            className="block py-3 text-base mobile:min-h-[44px]"
                        >
                            Companies
                        </Link>
                        <Link 
                            to="/technologies"
                            className="block py-3 text-base mobile:min-h-[44px]"
                        >
                            Technologies
                        </Link>
                        
                        {/* Theme Toggle */}
                        <div className="pt-4 border-t border-border">

export function CompanyProfilePage() {
    const { companyId } = useParams();
    const navigate = useNavigate();
    
    // Fix: Handle direct navigation with proper error state
    const { data, loading, error } = useQuery(() => 
        api.getCompanyProfile(companyId!)
    );

    if (loading) return <PageLoader />;
    
    if (error || !data) {
        // Better error handling for mobile
        return (
            <div className="min-h-screen flex items-center justify-center p-4">
                <ErrorState
                    title="Company Not Found"
                    message={`We couldn't find the company "${companyId}"`}
                    action={
                        <Button onClick={() => navigate('/companies')}>
                            Browse Companies
                        </Button>
                    }
                />
            </div>
        );
    }

    return (
        <div className="pb-20">  {/* Extra padding for mobile bottom nav if added */}
            <PageHeader 
                title={data.name}
                subtitle={`${data.industries.join(', ')} • ${data.operatingCountries.join(', ')}`}
                backLink="/companies"
            />

            {/* Content - Ensure scrollable on mobile */}
            <div className="overflow-x-hidden">
                {/* Stats - Stack on mobile */}
                <div className="grid grid-cols-2 md:grid-cols-4 gap-4 mb-6">
                    <StatCard label="Active Jobs" value={data.activeJobs} />
                    <StatCard label="Technologies" value={data.techStack.length} />
                    <StatCard label="Locations" value={data.hiringLocations.length} />
                    <StatCard label="Founded" value={data.foundedYear || 'N/A'} />

                    {activeTab === 'tech' && <TechStack techs={data.techStack} />}

            </div>
        </div>

### Step 6: Add Mobile-Specific CSS Utilities (1 hour)

**File:** `frontend/src/index.css`

```css
@layer utilities {
    /* Hide scrollbar but keep functionality */
    .mobile:no-scrollbar {
        -ms-overflow-style: none;  /* IE and Edge */
        scrollbar-width: none;  /* Firefox */
    }
    
    .mobile:no-scrollbar::-webkit-scrollbar {
        display: none;  /* Chrome, Safari, Opera */
    }

    /* Larger tap targets for mobile */
    @media (max-width: 768px) {
        .mobile\:min-h-\[44px\] {
            min-height: 44px;
        }
        
        .mobile\:min-w-\[44px\] {
            min-width: 44px;
        }
        
        .mobile\:max-w-\[280px\] {
            max-width: 280px;
        }
        
        .mobile\:hover\:shadow-none:hover {
            box-shadow: none;
        }
    }
}

@layer base {
    /* Ensure touch targets are large enough */
    @media (max-width: 768px) {
        button, a, input, select {
            min-height: 44px;
        }
    }

    /* Prevent text overflow */
    h1, h2, h3, h4, h5, h6 {
        word-wrap: break-word;
        overflow-wrap: break-word;
    }
}
```
