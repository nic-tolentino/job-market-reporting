## Executive Summary

### Problem
The current single `companies.json` file (2,500+ lines, 180+ companies) is becoming a bottleneck:
- **Merge conflicts** when multiple contributors edit simultaneously
- **Large diffs** make PR review difficult
- **Single point of failure** - syntax error breaks entire manifest
- **Hard to navigate** - finding specific company requires scrolling

### Solution
Migrate to directory-based structure where each company has its own JSON file:
```
data/
  companies/
    xero.json
    canva.json
    rocket-lab.json
    ...
```

### Impact
- Zero merge conflicts for parallel work
- Safer edits (one file = one company)
- Clean, reviewable diffs
- Easier to maintain and extend

### Step 2: Create Migration Script (2 hours)

**File:** `scripts/companies/migrate_to_directory.py`

```python
#!/usr/bin/env python3
"""
Migration script to split companies.json into individual files.

Usage:
    python3 scripts/companies/migrate_to_directory.py

This script will:
1. Read data/companies.json
2. Create data/companies/{company-id}.json for each company
3. Sort files alphabetically by company ID
4. Verify all files are valid JSON
5. Report migration statistics
"""

import json
import os
from pathlib import Path
from typing import List, Dict, Any

# Paths
PROJECT_ROOT = Path(__file__).parent.parent.parent
LEGACY_FILE = PROJECT_ROOT / "data" / "companies.json"
OUTPUT_DIR = PROJECT_ROOT / "data" / "companies"

def load_companies() -> List[Dict[str, Any]]:
    """Load companies from legacy JSON file."""
    print(f"Loading companies from {LEGACY_FILE}...")
    
    with open(LEGACY_FILE, 'r', encoding='utf-8') as f:
        companies = json.load(f)
    
    print(f"Loaded {len(companies)} companies")
    return companies

def validate_company(company: Dict[str, Any]) -> bool:
    """Validate company has required fields."""
    required_fields = ['id', 'name', 'verification_level']
    
    for field in required_fields:
        if field not in company:
            print(f"  ⚠️  Company missing required field '{field}': {company.get('id', 'UNKNOWN')}")
            return False
    
    # Validate ID is slugified
    company_id = company['id']
    if not company_id.replace('-', '').replace('_', '').isalnum():
        print(f"  ⚠️  Company ID contains invalid characters: {company_id}")
        return False
    
    return True

def save_company(company: Dict[str, Any], output_dir: Path) -> Path:
    """Save company to individual JSON file."""
    company_id = company['id']
    output_file = output_dir / f"{company_id}.json"
    
    # Sort keys for consistent formatting
    sorted_company = dict(sorted(company.items()))
    
    with open(output_file, 'w', encoding='utf-8') as f:
        json.dump(sorted_company, f, indent=2, ensure_ascii=False)
        f.write('\n')  # Trailing newline
    
    return output_file

def migrate():
    """Run the migration."""
    print("🚀 Starting migration to directory-based manifest...\n")
    
    # Ensure output directory exists
    OUTPUT_DIR.mkdir(parents=True, exist_ok=True)
    print(f"✓ Output directory: {OUTPUT_DIR}")
    
    # Load companies
    companies = load_companies()
    
    # Migrate each company
    successful = 0
    failed = 0
    verification_levels = {}
    
    print(f"\nMigrating companies...")
    for company in companies:
        if not validate_company(company):
            failed += 1
            continue
        
        try:
            output_file = save_company(company, OUTPUT_DIR)
            successful += 1
            
            # Track verification levels
            level = company.get('verification_level', 'unknown')
            verification_levels[level] = verification_levels.get(level, 0) + 1
            
            # Progress indicator
            if successful % 50 == 0:
                print(f"  Migrated {successful} companies...")
                
        except Exception as e:
            print(f"  ❌ Error migrating {company.get('id', 'UNKNOWN')}: {e}")
            failed += 1
    
    # Print summary
    print("\n" + "="*60)
    print("📊 Migration Summary")
    print("="*60)
    print(f"✓ Successfully migrated: {successful} companies")
    print(f"✗ Failed: {failed} companies")
    print(f"\nVerification Levels:")
    for level, count in sorted(verification_levels.items()):
        print(f"  - {level}: {count}")
    print(f"\nOutput directory: {OUTPUT_DIR}")
    print(f"Files created: {len(list(OUTPUT_DIR.glob('*.json')))}")
    print("="*60)
    
    if failed > 0:
        print(f"\n⚠️  {failed} companies failed validation. Please review and fix manually.")
        return 1
    
    print("\n✅ Migration completed successfully!")
    print("\nNext steps:")
    print("1. Review the migrated files in data/companies/")
    print("2. Update backend CompanySyncService to read from directory")
    print("3. Update enrich_unverified_companies.py to write individual files")
    print("4. Test thoroughly in staging")
    print("5. Delete legacy data/companies.json")
    
    return 0

if __name__ == "__main__":
    exit(migrate())
```

@Service
class CompanySyncService {
    @Value("\${app.companies.manifest-dir:data/companies}")
    private val manifestDir: String = "data/companies"

    private val log = LoggerFactory.getLogger(CompanySyncService::class.java)

    fun syncFromManifest(): List<CompanyRecord> {
        val companies = loadCompaniesFromDirectory()
        log.info("Loaded ${companies.size} companies from manifest directory")
        
        // Rest of sync logic remains the same
        return processCompanies(companies)
    }

    private fun loadCompaniesFromDirectory(): List<CompanyManifestDto> {
        val dir = File(manifestDir)
        
        if (!dir.exists() || !dir.isDirectory) {
            throw IllegalStateException("Company manifest directory not found: $manifestDir")
        }

        val jsonFiles = dir.listFiles { file -> 
            file.isFile && file.extension == "json" && !file.name.startsWith(".")
        } ?: emptyArray()

        if (jsonFiles.isEmpty()) {
            log.warn("No company JSON files found in $manifestDir")
            return emptyList()
        }

        return jsonFiles
            .sortedBy { it.name }  // Alphabetical order
            .mapNotNull { file ->
                try {
                    objectMapper.readValue<CompanyManifestDto>(file)
                        .copy(sourceFile = file.name)  // Track source for debugging
                } catch (e: Exception) {
                    log.error("Failed to parse company file ${file.name}: ${e.message}")
                    null
                }
            }
    }

    data class CompanyManifestDto(
        val id: String,
        val name: String,
        val alternateNames: List<String> = emptyList(),
        val description: String? = null,
        val website: String? = null,
        val logoUrl: String? = null,
        val industries: List<String> = emptyList(),
        val companyType: String? = null,
        val isAgency: Boolean = false,
        val isSocialEnterprise: Boolean = false,
        val hqCountry: String,
        val operatingCountries: List<String> = emptyList(),
        val officeLocations: List<String> = emptyList(),
        val remotePolicy: String? = null,
        val visaSponsorship: Boolean? = null,
        val employeesCount: Int? = null,
        val verificationLevel: String,
        val updatedAt: Instant? = null,
        // Internal tracking (not persisted to JSON)
        val sourceFile: String? = null
    )
}

### Step 5: Add Validation & CI Checks (1 hour)

**File:** `scripts/companies/validate_manifest.py`

```python
#!/usr/bin/env python3
"""
Validate company manifest files.

Usage:
    python3 scripts/companies/validate_manifest.py

Checks:
- All JSON files are valid
- Required fields present
- IDs are properly slugified
- No duplicate company IDs
- Verification levels are valid
"""

import json
import sys
from pathlib import Path
from typing import List, Dict, Any, Set

MANIFEST_DIR = Path(__file__).parent.parent.parent / "data" / "companies"

REQUIRED_FIELDS = ['id', 'name', 'verification_level']
VALID_VERIFICATION_LEVELS = ['verified', 'community_verified', 'silver', 'unverified', 'needs_review', 'blocked']

def validate_file(file: Path) -> bool:
    """Validate a single company JSON file."""
    errors = []
    
    try:
        with open(file, 'r', encoding='utf-8') as f:
            company = json.load(f)
    except json.JSONDecodeError as e:
        print(f"❌ {file.name}: Invalid JSON - {e}")
        return False
    
    # Check required fields
    for field in REQUIRED_FIELDS:
        if field not in company:
            errors.append(f"Missing required field: {field}")
    
    # Validate ID format
    company_id = company.get('id', '')
    if not company_id:
        errors.append("Company ID is empty")
    elif not company_id.replace('-', '').replace('_', '').isalnum():
        errors.append(f"Invalid characters in ID: {company_id}")
    
    # Validate verification level
    level = company.get('verification_level')
    if level and level not in VALID_VERIFICATION_LEVELS:
        errors.append(f"Invalid verification level: {level}")
    
    # Validate HQ country
    hq = company.get('hq_country')
    if hq and len(hq) != 2:
        errors.append(f"HQ country should be 2-letter ISO code: {hq}")
    
    # Report results
    if errors:
        print(f"❌ {file.name}:")
        for error in errors:
            print(f"   - {error}")
        return False
    else:
        print(f"✓ {file.name}")
        return True

def check_duplicates(files: List[Path]) -> bool:
    """Check for duplicate company IDs."""
    ids: Set[str] = set()
    duplicates = []
    
    for file in files:
        with open(file, 'r', encoding='utf-8') as f:
            company = json.load(f)
            company_id = company.get('id', '')
            
            if company_id in ids:
                duplicates.append(company_id)
            else:
                ids.add(company_id)
    
    if duplicates:
        print(f"\n❌ Duplicate company IDs found: {', '.join(duplicates)}")
        return False
    else:
        print(f"\n✓ No duplicate IDs ({len(ids)} unique companies)")
        return True

def validate():
    """Run all validations."""
    print("🔍 Validating company manifest files...\n")
    
    if not MANIFEST_DIR.exists():
        print(f"❌ Manifest directory not found: {MANIFEST_DIR}")
        return 1
    
    json_files = sorted([f for f in MANIFEST_DIR.glob('*.json') if not f.name.startswith('.')])
    
    if not json_files:
        print(f"❌ No JSON files found in {MANIFEST_DIR}")
        return 1
    
    print(f"Found {len(json_files)} company files\n")
    
    # Validate each file
    valid_count = 0
    for file in json_files:
        if validate_file(file):
            valid_count += 1
    
    # Check for duplicates
    if not check_duplicates(json_files):
        return 1
    
    # Summary
    print(f"\n{'='*60}")
    print(f"Validation Results: {valid_count}/{len(json_files)} files valid")
    print(f"{'='*60}")
    
    if valid_count != len(json_files):
        print(f"\n⚠️  {len(json_files) - valid_count} files have errors")
        return 1
    
    print("\n✅ All manifest files are valid!")
    return 0

if __name__ == "__main__":
    sys.exit(validate())
```

## Appendix: File Naming Convention

**Company ID Slugification Rules:**

1. Lowercase all letters
2. Replace spaces with hyphens
3. Remove special characters (except hyphens)
4. Remove consecutive hyphens
5. Trim leading/trailing hyphens

**Examples:**

| Company Name | Company ID | Filename |
|--------------|------------|----------|
| Xero | `xero` | `xero.json` |
