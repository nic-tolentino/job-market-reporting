## Executive Summary

### Problem: "Ghost Jobs"
Job postings that are no longer active but still appear on our platform:
1. Job is filled but company doesn't remove posting
2. Apply URL returns 404/410
3. URL redirects to generic careers page
4. Job requires login to view (effectively closed)

**Impact:** Users waste time applying to closed positions, eroding trust in platform data quality.

### Solution
**Daily Health Check Worker:**
- Scheduled job runs daily at 2:00 AM UTC
- Sends HEAD/GET requests to all active job apply URLs
- Analyzes response status and content
- Marks jobs as `CLOSED` or `UNVERIFIED` based on results
- Generates health report and alerts if >10% failures

### Impact
- Zero ghost jobs on platform
- High candidate trust
- Accurate job counts for companies/technologies

```kotlin
@Component
class HttpHealthChecker(private val webClient: WebClient) {

    data class HealthCheckResult(
        val url: String,
        val status: UrlStatus,
        val httpStatusCode: Int?,
        val redirectUrl: String?,
        val failureReason: String?,
        val responseTimeMs: Long
    )

    enum class UrlStatus {
        ACTIVE,
        CLOSED_404,
        CLOSED_410,
        CLOSED_REDIRECT,
        CLOSED_NO_LONGER,
        CLOSED_FILLED,
        UNVERIFIED_LOGIN,
        UNVERIFIED_TIMEOUT,
        UNVERIFIED_ERROR
    }

    suspend fun checkUrl(url: String, jobTitle: String?): HealthCheckResult {
        val startTime = System.currentTimeMillis()

        try {
            // Try HEAD request first
            val headResponse = webClient.method(HttpMethod.HEAD)
                .uri(url)
                .header("User-Agent", "DevAssembly-HealthChecker/1.0")
                .retrieve()
                .toBodilessEntity()
                .block(Duration.ofSeconds(10))

            val statusCode = headResponse?.statusCode?.value()
            val location = headResponse?.headers?.location?.toString()

            return when {
                statusCode == 200 -> verifyContentWithGet(url, jobTitle, startTime)
                statusCode == 404 -> HealthCheckResult(url, UrlStatus.CLOSED_404, statusCode, null, "404 Not Found", System.currentTimeMillis() - startTime)
                statusCode == 410 -> HealthCheckResult(url, UrlStatus.CLOSED_410, statusCode, null, "410 Gone", System.currentTimeMillis() - startTime)
                statusCode in 300..399 -> {
                    if (isGenericCareersPage(location)) {
                        HealthCheckResult(url, UrlStatus.CLOSED_REDIRECT, statusCode, location, "Redirects to generic careers page", System.currentTimeMillis() - startTime)
                    } else {
                        followRedirect(url, location, jobTitle, startTime)
                    }
                }
                else -> HealthCheckResult(url, UrlStatus.UNVERIFIED_ERROR, statusCode, null, "Unknown status", System.currentTimeMillis() - startTime)
            }
        } catch (e: Exception) {
            return handleException(e, url, startTime)
        }
    }
```

### Step 3: Create Health Check Service (2 hours)

**File:** `backend/src/main/kotlin/com/techmarket/service/JobHealthCheckService.kt`

```kotlin
@Service
class JobHealthCheckService(
    private val jobRepository: JobRepository,
    private val httpHealthChecker: HttpHealthChecker,
    private val notificationService: NotificationService
) {

    suspend fun runHealthCheck(): HealthCheckSummary {
        val jobsToCheck = jobRepository.getJobsNeedingHealthCheck(limit = 1000)
        log.info("Starting health check for ${jobsToCheck.size} jobs")

        val results = mutableListOf<HealthCheckResult>()
        val semaphore = Semaphore(20)  // Max 20 concurrent requests

        coroutineScope {
            jobsToCheck.map { job ->
                launch {
                    semaphore.acquire()
                    try {
                        val result = httpHealthChecker.checkUrl(
                            url = job.applyUrl ?: return@launch,
                            jobTitle = job.title
                        )
                        results.add(result)
                        updateJobHealth(job.id, result)
                    } finally {
                        semaphore.release()
                    }
                }
            }
        }

        val summary = HealthCheckSummary(
            totalChecked = results.size,
            activeCount = results.count { it.status == UrlStatus.ACTIVE },
            closedCount = results.count { it.status.name.startsWith("CLOSED_") },
            unverifiedCount = results.count { it.status.name.startsWith("UNVERIFIED_") },
            avgResponseTimeMs = results.map { it.responseTimeMs }.average()
        )

        // Alert if high failure rate
        val failureRate = (summary.closedCount + summary.unverifiedCount).toDouble() / summary.totalChecked
        if (failureRate > 0.1) {
            notificationService.sendAlert("⚠️ High job URL failure rate: ${(failureRate * 100).toInt()}%")
        }

        return summary
    }
}
```

### Step 4: Create Scheduler (1 hour)

```kotlin
@Component
class HealthCheckScheduler(private val healthCheckService: JobHealthCheckService) {

    @Scheduled(cron = "0 0 2 * * *", zone = "UTC")  // 2:00 AM UTC
    fun runDailyHealthCheck() {
        log.info("Starting scheduled daily health check")
        try {
            runBlocking {
                val summary = healthCheckService.runHealthCheck()
                log.info("Daily health check completed: $summary")
            }
        } catch (e: Exception) {
            log.error("Daily health check failed: ${e.message}", e)
        }
    }
}
```
