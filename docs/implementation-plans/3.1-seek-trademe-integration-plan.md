## Executive Summary

### Problem
LinkedIn-only sourcing misses **50-60% of ANZ tech jobs**. SEEK dominates AU (~90% job seeker time), TradeMe significant in NZ.

### Solution
Use Apify Store scrapers for SEEK and TradeMe (don't build custom):
- Create separate Apify tasks for each job board
- Implement deduplication logic
- Prefer direct ATS links over job board links

### Impact
- **Double job coverage** (50-60% more jobs)
- **Better market representation**
- **Reduced dependency on LinkedIn**

### Step 3: Implement Deduplication Logic (3 hours)

**File:** `backend/src/main/kotlin/com/techmarket/sync/JobDeduplicator.kt`

```kotlin
@Service
class JobDeduplicator(private val jobRepository: JobRepository) {

    data class DedupKey(
        val companyId: String,
        val title: String,
        val location: String
    )

    fun findDuplicates(newJobs: List<JobRecord>): List<JobRecord> {
        val existingJobs = jobRepository.getActiveJobs()
        val existingKeys = existingJobs.associateBy { 
            DedupKey(it.companyId, it.title.normalize(), it.location.normalize()) 
        }

        return newJobs.filter { newJob ->
            val key = DedupKey(
                newJob.companyId,
                newJob.title.normalize(),
                newJob.location.normalize()
            )
            existingKeys.containsKey(key)
        }
    }

    private fun String.normalize(): String {
        return lowercase()
            .replace(Regex("[^a-z0-9]"), " ")
            .trim()
            .replace(Regex("\\s+"), " ")
    }
}
```

**Priority Logic:**
```
ATS Direct > SEEK/TradeMe > LinkedIn
```

### Step 4: Update Sync Service (3 hours)

**File:** `backend/src/main/kotlin/com/techmarket/sync/JobDataSyncService.kt`

```kotlin
@Service
class JobDataSyncService {

    suspend fun runDataSync(source: String = "ALL") {
        when (source) {
            "SEEK" -> syncSeekJobs()
            "TradeMe" -> syncTradeMeJobs()
            "LinkedIn" -> syncLinkedInJobs()
            "ALL" -> {
                syncLinkedInJobs()
                syncSeekJobs()
                syncTradeMeJobs()
            }
        }
    }

    private suspend fun syncSeekJobs() {
        val seekJobs = apifyClient.getSeekJobs()
        val deduplicated = deduplicate(seekJobs)
        jobRepository.upsertJobs(deduplicated)
    }

    private suspend fun syncTradeMeJobs() {
        val trademeJobs = apifyClient.getTradeMeJobs()
        val deduplicated = deduplicate(trademeJobs)
        jobRepository.upsertJobs(deduplicated)
    }
}
```
