## Executive Summary

### Problem
Salary data from job postings is highly inconsistent across different sources and countries:
- **Different currencies:** NZD, AUD, USD, EUR
- **Different periods:** Hourly, daily, monthly, annually
- **Different formats:** "$120,000", "120k", "$120k-$150k", "€35.000"
- **Different contexts:** Base salary vs total compensation, gross vs net
- **Country-specific formats:** European format (35.000€) vs US/AU/NZ format ($35,000)

This makes it impossible to compare salaries across job postings or provide meaningful salary insights.

### Solution
Create a `NormalizedSalary` data structure that captures amount, currency, period, source, and gross/net status. Implement robust parsing logic that handles all regional formats and provides confidence levels based on data source.

### Impact
- Enables salary comparison across all job postings
- Provides transparency with confidence indicators
- Supports multi-country expansion (AU, NZ, ES)
- Foundation for future salary insights and filtering features

```kotlin
package com.techmarket.model

/**
 * Represents a normalized salary value with currency, period, and source information.
 *
 * @property amount The salary amount in the smallest currency unit (e.g., cents)
 * @property currency ISO 4217 currency code (NZD, AUD, USD, EUR)
 * @property period The time period for the salary (HOUR, DAY, MONTH, YEAR)
 * @property source The source of the salary data (determines confidence level)
 * @property isGross Whether this is gross salary (true) or net (false). Defaults to true.
 * @property disclaimer Optional disclaimer for low-confidence estimates
 */
data class NormalizedSalary(
    val amount: Long,
    val currency: String,      // NZD, AUD, USD, EUR
    val period: String,        // HOUR, DAY, MONTH, YEAR
    val source: String,        // JOB_POSTING, ATS_API, MARKET_DATA, AI_ESTIMATE
    val isGross: Boolean = true,
    val disclaimer: String? = null
) {
    companion object {
        // Source constants
        const val SOURCE_JOB_POSTING = "JOB_POSTING"    // HIGH confidence
        const val SOURCE_ATS_API = "ATS_API"            // HIGH confidence
        const val SOURCE_MARKET_DATA = "MARKET_DATA"    // MEDIUM confidence
        const val SOURCE_AI_ESTIMATE = "AI_ESTIMATE"    // LOW confidence

        // Period constants
        const val PERIOD_HOUR = "HOUR"
        const val PERIOD_DAY = "DAY"
        const val PERIOD_MONTH = "MONTH"
        const val PERIOD_YEAR = "YEAR"

        // Currency constants
        const val CURRENCY_NZD = "NZD"
        const val CURRENCY_AUD = "AUD"
        const val CURRENCY_USD = "USD"
        const val CURRENCY_EUR = "EUR"
        const val CURRENCY_GBP = "GBP"

        fun getConfidenceForSource(source: String): String = when (source) {
            SOURCE_JOB_POSTING -> "HIGH"
            SOURCE_ATS_API -> "HIGH"
            SOURCE_MARKET_DATA -> "MEDIUM"
            SOURCE_AI_ESTIMATE -> "LOW"
            else -> "UNKNOWN"
        }

        fun getDefaultPeriodForCountry(country: String): String = when (country.uppercase()) {
            "NZ", "AU", "US", "UK", "ES", "CA", "IE" -> PERIOD_YEAR
            else -> PERIOD_YEAR
        }

        fun getDefaultCurrencyForCountry(country: String): String = when (country.uppercase()) {
            "NZ" -> CURRENCY_NZD
            "AU" -> CURRENCY_AUD
            "US" -> CURRENCY_USD
            "ES", "DE", "FR", "IT", "IE", "NL", "BE" -> CURRENCY_EUR
            "UK" -> "GBP"
            "CA" -> "CAD"
            else -> CURRENCY_NZD
        }
    }

    fun toAnnualAmount(): Long = when (period) {
        PERIOD_HOUR -> amount * 2080  // 40 hrs/week × 52 weeks
        PERIOD_DAY -> amount * 260    // 5 days/week × 52 weeks
        PERIOD_MONTH -> amount * 12
        PERIOD_YEAR -> amount
        else -> amount
    }

    val confidence: String
        get() = getConfidenceForSource(source)
}
```

**Key Parsing Rules:**

| Pattern | Currency | Period | Notes |
|---------|----------|--------|-------|
| `$` + country=AU | AUD | YEAR | Default to annual |
| `$` + country=NZ | NZD | YEAR | Default to annual |
| `€` or `EUR` | EUR | YEAR | European format |
| `/hr`, `/hour`, `p/h` | Detected | HOUR | Explicit hourly |
| `/day`, `per day` | Detected | DAY | Daily rate |
| `/month`, `pcm` | Detected | MONTH | Monthly |
| `k` suffix (e.g., 80k) | Detected | YEAR | Multiply by 1000 |
| European format (35.000€) | EUR | YEAR | Dot = thousands separator |
| `$120k + super` | Detected | YEAR | Extract base only |
| `Salario Bruto 35.000€` | EUR | YEAR | Spanish "Bruto" = gross |
| `Salario Neto 30.000€` | EUR | YEAR | Spanish "Neto" = net |

**European Number Format Handling:**

```kotlin
private fun extractAmount(salaryStr: String, currency: String): Long {
    // Determine thousands separator based on currency/country
    val thousandsSeparator = when (currency) {
        "EUR" -> "."  // Most EU uses dot for thousands
        else -> ","   // AU/NZ/US use comma
    }

    val decimalSeparator = when (currency) {
        "EUR" -> ","  // Most EU uses comma for decimals
        else -> "."   // AU/NZ/US use dot
    }

    // Remove currency symbols and whitespace
    var cleaned = salaryStr
        .replace(Regex("[€$£]"), "")
        .replace("EUR", "")
        .replace("USD", "")
        .trim()

    // Handle "k" suffix
    if (cleaned.endsWith("k", ignoreCase = true)) {
        cleaned = cleaned.dropLast(1) + "000"
    }

    // Stop at "+" sign (e.g., "$120k + super" → "$120k")
    cleaned = cleaned.substringBefore("+").trim()

    // Replace thousands separator
    cleaned = cleaned.replace(thousandsSeparator, "")

    // Extract digits only
    val digits = cleaned.replace(Regex("[^0-9]"), "")

    return if (digits.isNotBlank()) {
        digits.toLong() * 100  // Convert to cents
    } else {
        0L
    }
}
```

```typescript
export interface NormalizedSalary {
    amount: number;       // In cents
    currency: string;     // NZD, AUD, USD, EUR
    period: string;       // HOUR, DAY, MONTH, YEAR
    source: string;       // JOB_POSTING, ATS_API, MARKET_DATA, AI_ESTIMATE
    isGross: boolean;
    disclaimer?: string | null;
}

export function formatSalary(salary: NormalizedSalary, locale: string = 'en-NZ'): string {
    const formatter = new Intl.NumberFormat(locale, {
        style: 'currency',
        currency: salary.currency,
        maximumFractionDigits: 0,
    });

    const amount = salary.amount / 100;  // Convert from cents
    const period = formatPeriod(salary.period);

    return `${formatter.format(amount)}${period}`;
}

export function formatSalaryRange(
    min: NormalizedSalary,
    max: NormalizedSalary,
    locale: string = 'en-NZ'
): string {
    const minFormatted = formatSalary(min, locale);
    const maxFormatted = formatSalary(max, locale);
    return `${minFormatted} - ${maxFormatted}`;
}

export function getConfidenceColor(source: string): string {
    switch (source) {
        case 'JOB_POSTING':
        case 'ATS_API':
            return 'text-green-600 bg-green-50 dark:bg-green-900/20 dark:text-green-400';
        case 'MARKET_DATA':
            return 'text-yellow-600 bg-yellow-50 dark:bg-yellow-900/20 dark:text-yellow-400';
        case 'AI_ESTIMATE':
            return 'text-orange-600 bg-orange-50 dark:bg-orange-900/20 dark:text-orange-400';
        default:
            return 'text-gray-600 bg-gray-50 dark:bg-gray-900/20 dark:text-gray-400';
    }
}

export function getConfidenceLabel(source: string): string {
    switch (source) {
        case 'JOB_POSTING':
        case 'ATS_API':
            return 'Verified';
        case 'MARKET_DATA':
            return 'Estimated';
        case 'AI_ESTIMATE':
            return 'AI Estimate';
        default:
            return 'Unknown';
    }
}

function formatPeriod(period: string): string {
    switch (period) {
        case 'HOUR': return '/hr';
        case 'DAY': return '/day';
        case 'MONTH': return '/month';
        case 'YEAR': return '';
        default: return '';
    }
}
```

## Edge Cases & Mitigations

| Edge Case | Mitigation |
|-----------|------------|
| Salary range without currency | Infer currency from job country |
| "Competitive salary" or "Market rate" | Return null (no structured data) |
| Superannuation included | Parse base only, stop at "+" |
| Package deals | Parse base, mark source=MEDIUM |
| Multiple currencies in one posting | Use primary currency, log warning |
| Ambiguous period | Use country-specific default |
| European number format | Use dot as thousands separator for EUR |
| Spanish "Bruto" vs "Neto" | Detect keywords, set isGross flag |
| Salary in crypto/equity only | Return null, note in benefits field |
