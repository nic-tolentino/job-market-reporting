## Executive Summary

### Problem
Native iOS developers see Xamarin/React Native jobs when filtering for iOS - causes frustration and noise.

### Solution
Add technology exclusion logic at query level:
- API: `GET /api/tech/iOS?exclude=Xamarin,ReactNative,Flutter`
- UI: "Native Only" toggle on relevant tech pages

### Impact
- Cleaner search results
- Better user experience for native developers
- More accurate technology matching

### Step 1: Add Exclusion Logic to TechFormatter (1 hour)

**File:** `backend/src/main/kotlin/com/techmarket/util/TechFormatter.kt`

```kotlin
object TechFormatter {
    val defaultExclusions = mapOf(
        "iOS" to listOf("Xamarin", "React Native", "Flutter", "Ionic"),
        "Android" to listOf("Xamarin", "React Native", "Flutter", "Ionic"),
        "Java" to listOf("Kotlin"),  // For "Pure Java" roles
        "Kotlin" to listOf("Java")   // For "Pure Kotlin" roles
    )

    fun getExclusionsForTech(tech: String): List<String> {
        return defaultExclusions[tech] ?: emptyList()
    }

    fun isCrossPlatform(tech: String): Boolean {
        return listOf("Xamarin", "React Native", "Flutter", "Ionic").contains(tech)
    }
}
```

### Step 2: Update API Controller (1 hour)

**File:** `backend/src/main/kotlin/com/techmarket/api/TechController.kt`

```kotlin
@RestController
@RequestMapping("/api/tech")
class TechController(
    private val repository: JobRepository
) {

    @GetMapping("/{techName}")
    fun getTechPage(
        @PathVariable techName: String,
        @RequestParam exclude: List<String>? = null,
        @RequestParam nativeOnly: Boolean? = null,
        @RequestParam country: String? = null
    ): ResponseEntity<TechPageDto> {
        // Build exclusion list
        val exclusions = if (nativeOnly == true) {
            exclude.orEmpty() + TechFormatter.getExclusionsForTech(techName)
        } else {
            exclude.orEmpty()
        }

        val jobs = repository.getJobsForTechnology(
            tech = techName,
            exclusions = exclusions,
            country = country
        )

        val dto = TechPageDto(
            techName = TechFormatter.format(techName),
            totalJobs = jobs.size,
            jobs = jobs
        )

        return ResponseEntity.ok(dto)
    }
}
```

### Step 3: Update BigQuery Query (2 hours)

**File:** `backend/src/main/kotlin/com/techmarket/persistence/job/JobQueries.kt`

```kotlin
fun getJobsForTechnology(
    tech: String,
    exclusions: List<String>,
    country: String?
): List<JobRecord> {
    val baseQuery = """
        SELECT *
        FROM \`${JobFields.TABLE}\` j,
        UNNEST(j.technologies) as t
        WHERE LOWER(t) = LOWER(@tech)
        ${if (country != null) "AND j.country = @country" else ""}
    """.trimIndent()

    val exclusionClause = if (exclusions.isNotEmpty()) {
        """
        AND NOT EXISTS (
            SELECT 1
            FROM UNNEST(j.technologies) as t_ex
            WHERE LOWER(t_ex) IN UNNEST(@exclusions)
        )
        """.trimIndent()
    } else ""

    val query = baseQuery + exclusionClause

    return executeQuery(query, mapOf(
        "tech" to tech,
        "exclusions" to exclusions.map { it.lowercase() },
        "country" to country
    ))
}
```

### Step 4: Frontend - Add Native Only Toggle (2 hours)

**File:** `frontend/src/pages/TechDetailsPage.tsx`

```tsx
export function TechDetailsPage() {
    const { techName } = useParams();
    const [nativeOnly, setNativeOnly] = useState(false);

    const { data } = useQuery(() => 
        api.getTechPage(techName!, {
            nativeOnly: nativeOnly,
            exclude: nativeOnly ? TechFormatter.getExclusions(techName!) : []
        })
    );

    const showNativeToggle = ['iOS', 'Android'].includes(techName!);

    return (
        <div>
            <PageHeader title={TechFormatter.format(techName!)} />

            {showNativeToggle && (
                <div className="mb-6">
                    <Checkbox
                        checked={nativeOnly}
                        onChange={setNativeOnly}
                        label="🎯 Native Only (exclude cross-platform)"
                    />
                    {nativeOnly && (
                        <p className="text-sm text-gray-600 mt-1">
                            Excluding: {TechFormatter.getExclusions(techName!).join(', ')}
                        </p>
                    )}
                </div>
            )}

            <JobList jobs={data?.jobs || []} />
        </div>
    );
}
```

**File:** `frontend/src/lib/techFormatter.ts`

```typescript
export const TechFormatter = {
    exclusions: {
        'iOS': ['Xamarin', 'React Native', 'Flutter', 'Ionic'],
        'Android': ['Xamarin', 'React Native', 'Flutter', 'Ionic']
    },

    getExclusions(tech: string): string[] {
        return this.exclusions[tech as keyof typeof this.exclusions] || [];
    }
};
```
