## Problem Statement

**Migrant workers cannot easily identify companies that sponsor work visas.**

This is a critical pain point for:
- International students graduating from NZ/AU universities
- Skilled workers looking to relocate from overseas
- Current visa holders seeking to switch employers

**Current State (as of March 2026):**
- Each company has its own JSON file in `data/companies/{letter}/{id}.json` — 1,258 files total
- Each file already has a `visa_sponsorship: boolean` field (e.g., `"visa_sponsorship": true`)
- The field flows through the full backend pipeline:
  - BigQuery `raw_companies.visa_sponsorship` (BOOL)
  - `CompanyRow.visaSponsorship: Boolean`
  - `CompanyRecord.visaSponsorship: Boolean`
  - `CompanyDetailsDto.visaSponsorship: Boolean`
- ✅ **Backend is complete** — the API already returns `visaSponsorship: boolean`
- ❌ **Frontend is missing the field** — `CompanyDetailsDto` in `frontend/src/lib/api.ts` doesn't include it
- ❌ **No UI display or filtering** exists

---

## Solution Overview

**Phase A (quick win — ~1-2 hours): Surface the existing boolean**

1. Add `visaSponsorship: boolean` to the frontend `CompanyDetailsDto` type
2. Display a visa sponsorship badge on company profiles

**Phase B (enhancement — ~1-2 hours): Enrich data with object schema**

1. Migrate `visa_sponsorship` from boolean to a structured object in company JSON files
2. Update backend to parse the new format (custom Jackson deserializer)
3. Migrate BigQuery schema from `BOOL` to `JSON`
4. Add a company listing endpoint with filtering
5. Build a `CompaniesPage` with visa sponsorship filter

### Step B2: Add Custom Jackson Deserializer for Mixed Format (1 hour)

**File:** `backend/src/main/kotlin/com/techmarket/sync/model/CompanyJsonDto.kt`

**New file:** `backend/src/main/kotlin/com/techmarket/sync/model/VisaSponsorshipDeserializer.kt`

```kotlin
package com.techmarket.sync.model

import com.fasterxml.jackson.core.JsonParser
import com.fasterxml.jackson.databind.DeserializationContext
import com.fasterxml.jackson.databind.JsonDeserializer
import com.fasterxml.jackson.databind.JsonToken

class VisaSponsorshipDeserializer : JsonDeserializer<VisaSponsorshipInfo?>() {
    override fun deserialize(p: JsonParser, ctxt: DeserializationContext): VisaSponsorshipInfo? {
        return when (val token = p.currentToken()) {
            JsonToken.VALUE_TRUE -> VisaSponsorshipInfo(offered = true)
            JsonToken.VALUE_FALSE -> VisaSponsorshipInfo(offered = false)
            JsonToken.VALUE_NULL -> null
            JsonToken.START_OBJECT -> ctxt.readValue(p, VisaSponsorshipInfo::class.java)
            else -> throw ctxt.wrongTokenException(
                p, 
                VisaSponsorshipInfo::class.java, 
                token, 
                "Expected boolean or object for visa_sponsorship"
            )
        }
    }
}
```

**Update `CompanyJsonDto.kt`:**

Add new data class:

```kotlin
    val types: List<String> = emptyList(),

    val lastVerified: String? = null,
    val source: String? = null
)
```

Update the field:

```kotlin
}
```

### Step B7: Create Company Listing Page with Filter (1 hour)

**File:** `frontend/src/pages/CompaniesPage.tsx` *(new file)*

```tsx
import { useState, useEffect } from 'react';
import { Link } from 'react-router-dom';
import { Plane } from 'lucide-react';
import { Badge } from '../components/ui/Badge';
import CompanyLogo from '../components/common/CompanyLogo';

interface CompanyListingItem {
    id: string;
    name: string;
    logo: string;
    visaSponsorship: { offered: boolean } | null;
    activeRoles: number;
}

export default function CompaniesPage() {
    const { selectedCountry } = useAppStore();
    const [visaSponsorshipOnly, setVisaSponsorshipOnly] = useState(false);
    const [companies, setCompanies] = useState<CompanyListingItem[]>([]);
    const [isLoading, setIsLoading] = useState(true);

    useEffect(() => {
        setIsLoading(true);
        const url = new URL('/api/company/listing', window.location.origin);
        if (selectedCountry) url.searchParams.append('country', selectedCountry);
        if (visaSponsorshipOnly) url.searchParams.append('visaSponsorship', 'true');
        
        fetch(url.toString())
            .then(res => res.json())
            .then(setCompanies)
            .finally(() => setIsLoading(false));
    }, [selectedCountry, visaSponsorshipOnly]);

    return (
        <div className="space-y-6">
            <div className="flex items-center justify-between">
                <h1 className="text-2xl font-bold">Companies</h1>
                <label className="flex items-center gap-2 text-sm font-medium text-secondary cursor-pointer">
                    <input
                        type="checkbox"
                        checked={visaSponsorshipOnly}
                        onChange={(e) => setVisaSponsorshipOnly(e.target.checked)}
                        className="rounded"
                    />
                    Visa sponsors only
                </label>
            </div>
            
            {isLoading ? (
                <PageLoader />
            ) : (
                <div className="grid gap-4 md:grid-cols-2 lg:grid-cols-3">
                    {companies.map((company) => (
                        <Link 
                            key={company.id} 
                            to={`/company/${company.id}`}
                            className="block p-4 border border-border rounded-lg hover:border-accent transition-colors"
                        >
                            <div className="flex items-start gap-3">
                                <CompanyLogo 
                                    logoUrl={company.logo} 
                                    companyName={company.name}
                                    className="h-12 w-12 rounded-lg"
                                />
                                <div className="flex-1">
                                    <h3 className="font-semibold">{company.name}</h3>
                                    <p className="text-sm text-muted">
                                        {company.activeRoles} active role{company.activeRoles !== 1 ? 's' : ''}
                                    </p>
                                    {company.visaSponsorship?.offered && (
                                        <Badge variant="blue" icon={<Plane className="h-3 w-3" />}>
                                            Visa Sponsor
                                        </Badge>
                                    )}
                                </div>
                            </div>
                        </Link>
                    ))}
                </div>
            )}
        </div>
    );
}
```
