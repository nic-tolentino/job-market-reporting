## Executive Summary

### Problem
Users can't browse jobs by high-level categories (Web, Mobile, Cloud, Data) - only individual technologies.

### Solution
Create domain hub pages that aggregate related technologies:
- **Web Development:** React, Angular, Vue, Node.js, etc.
- **Mobile Development:** iOS, Android, React Native, Flutter
- **Cloud & DevOps:** AWS, GCP, Azure, Kubernetes, Docker
- **Data & AI:** Python, SQL, ML frameworks, BigQuery

### Impact
- Better content discovery
- SEO landing pages for each domain
- Market insights at category level

```kotlin
        val displayName: String,

        "react" to TechMetadata("React", TechCategory.FRONTEND, "#61DAFB"),

        "android" to TechMetadata("Android", TechCategory.MOBILE, "#3DDC84"),
        // ... add all technologies
    )

    fun format(tech: String): String {
        return techMetadataMap[tech.lowercase()]?.displayName ?: tech.capitalize()
    }

    fun getMetadata(tech: String): TechMetadata {
        return techMetadataMap[tech.lowercase()] 
            ?: TechMetadata(tech.capitalize(), TechCategory.LANGUAGES)
    }

    fun getCategory(tech: String): TechCategory {
        return getMetadata(tech).category
    }
```

### Step 3: Create Domain Hub Controller (3 hours)

**File:** `backend/src/main/kotlin/com/techmarket/api/DomainHubController.kt`

```kotlin
@RestController
@RequestMapping("/api/hubs")
class DomainHubController(
    private val repository: InsightsBigQueryRepository
) {

    @GetMapping("/{category}")
    fun getDomainHub(
        @PathVariable category: String,
        @RequestParam country: String? = null
    ): ResponseEntity<DomainHubDto> {
        val techCategory = TechCategory.valueOf(category.uppercase())
        
        val technologies = repository.getTechnologiesByCategory(techCategory, country)
        val jobs = repository.getJobsByCategory(techCategory, country)
        val companies = repository.getCompaniesByCategory(techCategory, country)
        val trends = repository.getCategoryTrends(techCategory, country)

        val dto = DomainHubDto(
            category = techCategory,
            totalJobs = jobs.size,
            activeCompanies = companies.size,
            technologies = technologies,
            topCompanies = companies.take(10),
            recentJobs = jobs.take(20),
            trends = trends
        )

        return ResponseEntity.ok(dto)
    }
}

data class DomainHubDto(
    val category: TechCategory,
    val totalJobs: Int,
    val activeCompanies: Int,
    val technologies: List<TechnologyDto>,
    val topCompanies: List<CompanyDto>,
    val recentJobs: List<JobRoleDto>,
    val trends: CategoryTrendsDto
)
```

```kotlin
fun getTechnologiesByCategory(category: TechCategory, country: String?): List<TechnologyDto> {
    val techInCategory = TechFormatter.getAllTechnologiesForCategory(category)
    
    val query = """
        SELECT 
            t.tech as technology,
            COUNT(DISTINCT j.job_id) as jobCount,
            COUNT(DISTINCT j.company_id) as companyCount
        FROM \`${JobFields.TABLE}\` j,
        UNNEST(j.technologies) as t
        WHERE t.tech IN UNNEST(@technologies)
        ${if (country != null) "AND j.country = @country" else ""}
        GROUP BY t.tech
        ORDER BY jobCount DESC
    """.trimIndent()
```

### Step 5: Create Frontend Hub Pages (4 hours)

**File:** `frontend/src/pages/DomainHubPage.tsx`

```tsx
export function DomainHubPage() {
    const { category } = useParams<{ category: string }>();
    const { data, loading } = useQuery(() => 
        api.getDomainHub(category as string)
    );

    if (loading) return <PageLoader />;
    if (!data) return <ErrorState />;

    return (
        <div className="max-w-7xl mx-auto">
            <PageHeader
                title={data.category.displayName}
                description={data.category.description}
            />

            {/* Market Pulse */}
            <div className="grid grid-cols-3 gap-6 mb-8">
                <StatCard label="Active Jobs" value={data.totalJobs} />
                <StatCard label="Companies Hiring" value={data.activeCompanies} />
                <StatCard label="Technologies" value={data.technologies.length} />
            </div>

            {/* Technologies Grid */}
            <Section title="Technologies">
                <div className="grid grid-cols-4 gap-4">
                    {data.technologies.map(tech => (
                        <TechCard key={tech.name} tech={tech} />
                    ))}
                </div>
            </Section>

            {/* Top Companies */}
            <Section title="Top Companies Hiring">
                <CompanyList companies={data.topCompanies} />
            </Section>

            {/* Recent Jobs */}
            <Section title="Recent Jobs">
                <JobList jobs={data.recentJobs} />
            </Section>
        </div>
    );
}
```
